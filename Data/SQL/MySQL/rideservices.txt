--MySQL table creation script for RideServices table

CREATE TABLE rideservice (
 rideservice_id INT AUTO_INCREMENT PRIMARY KEY,
 rideservice_name VARCHAR(100) NOT NULL,
 priceperkm DECIMAL(10, 2) NOT NULL
);
